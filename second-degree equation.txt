	escreva("Insira o valor de a:" )

	leia(c)
	
	se (a = 0) entao
		escreva("Não é uma equação de segundo grau")
	se nao
		delta <- sqr(b) - 4 * a * c
		se (delta < 0) então
	escreva("Não existe raiz real")
		se nao
			se (delta = 0) entao
				escreva("Existe uma raiz real")
				x1 <- (-b) / (2*a)
				escreva("A raiz é: ", x1)
			se nao
				se (delta >0) entao 
					escreva("Existem duas raizes reais")
					x1 <- (-b + sqrt(delta)) / (2*a)
					escreva("A raiz x1 é: ", x1)
					x2 <- (-b - sqrt(delta)) / (2*a)
					escreva("A raiz x2 é: ", x2)
